\section{Development Setup Instructions}

\subsection{Prerequisites}
\begin{itemize}
    \item Node.js (v18 or higher)
    \item PostgreSQL database
    \item npm or yarn package manager
\end{itemize}

\subsection{Database Setup}
\begin{enumerate}
    \item Create a new PostgreSQL database
    \item Run schema setup:
    \begin{itemize}
        \item Use provided \texttt{newschema.sql}, or
        \item Use Prisma migrations
    \end{itemize}
    \item (Optional) Import sample data from \texttt{sample\_data\_simplified/}
\end{enumerate}

\subsection{Backend Setup}
\begin{enumerate}
    \item Navigate to backend directory: \texttt{cd backend}
    \item Install dependencies: \texttt{npm install}
    \item Copy environment file: \texttt{cp .env.example .env}
    \item Configure database URL in \texttt{.env}
    \item Generate Prisma client: \texttt{npm run db:generate}
    \item Apply database migrations: \texttt{npm run db:push}
    \item Start development server: \texttt{npm run dev}
\end{enumerate}

\subsection{Frontend Setup}
\begin{enumerate}
    \item Navigate to frontend directory: \texttt{cd frontend}
    \item Install dependencies: \texttt{npm install}
    \item Start development server: \texttt{npm run dev}
    \item Access application at \texttt{http://localhost:3000}
\end{enumerate}
